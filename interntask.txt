import json
import urllib.request as ur
import urllib.request as ur1

def get_headlines(no_of_headlines):
    s = ur.urlopen("https://hacker-news.firebaseio.com/v0/topstories.json")
    top_story_ids=s.read()

    ids = json.loads(top_story_ids)[:no_of_headlines]
    headlines = []
    for story_id in ids:
        story_url = "https://hacker-news.firebaseio.com/v0/item/{0}.json".format(story_id)
        story = ur1.urlopen(story_url).read()
        story_json = json.loads(story)
        headlines.append(story_json["title"])
        headlines.appensd(story_json['by']
    return headlines
